const fs = require('fs'); const path = require('path'); // Notion 資料庫 ID 對照表 const DATABASE_IDS = { "QuoteCards": "2d8c43406e35806db3c9f3a7c7ec9ce8", "ScenarioCards": "2bbc43406e35801ab2f7f65cf464d8aa", "WorldEvent": "2bbc43406e3580039ecaf66d86bceab8", "Glossary": "2c5c43406e35800aa4e4ee3406589a18", "Modules": "2bbc43406e358011be7be8b5883d70a3", "Theory": "2bbc43406e358033b3e4d7171e0a813f" }; // 必填欄位對照表 const REQUIRED_FIELDS = { "QuoteCards": ["名稱（Name）", "語錄原句（Quote）"], "ScenarioCards": ["副本名稱（Name）"], "WorldEvent": ["事件名稱（EventTitle）", "事件類型（EventType）"], "Glossary": ["語氣名稱（tone_name）", "語氣定義（tone_definition）"], "Modules": ["Name"], "Theory": ["Name"] }; // Select 欄位的有效值 const VALID_SELECT_VALUES = { "WorldEvent": { "事件類型（EventType）": ["角色人格調整", "關係變化", "世界觀規則更新", "模組升級", "副本誕生"], "重要程度（Priority）": ["S", "A", "B", "C"] }, "ScenarioCards": { "類型（ScenarioType）": ["主線", "副線", "語氣試壓", "副本快爆", "日常"] }, "Glossary": { "風險等級（risk_level）": ["Safe", "Caution", "Restricted"] } }; function validateJSON(filepath) { console.log(`\n🔍 檢查檔案: ${path.basename(filepath)}`); console.log("=".repeat(60)); let hasErrors = false; let warnings = []; try { // 讀取檔案 const content = fs.readFileSync(filepath, 'utf8'); const data = JSON.parse(content); // 檢查 1: database 欄位 if (!data.database) { console.log("❌ 錯誤: 缺少 'database' 欄位"); hasErrors = true; return; } console.log(`✅ 資料庫類型: ${data.database}`); // 檢查 2: database_id if (!data.database_id) { console.log("❌ 錯誤: 缺少 'database_id' 欄位"); hasErrors = true; } else if (data.database_id.includes("你的") || data.database_id.length < 20) { console.log(`❌ 錯誤: database_id 是占位符: ${data.database_id}`); hasErrors = true; } else { const expectedId = DATABASE_IDS[data.database]; if (expectedId && data.database_id !== expectedId) { console.log(`⚠️ 警告: database_id 可能不正確`); console.log(` 預期: ${expectedId}`); console.log(` 實際: ${data.database_id}`); warnings.push("database_id 不符預期"); } else { console.log(`✅ database_id: ${data.database_id.substring(0, 20)}...`); } } // 檢查 3: entries if (!data.entries || !Array.isArray(data.entries)) { console.log("❌ 錯誤: 缺少 'entries' 陣列"); hasErrors = true; return; } console.log(`✅ 條目數量: ${data.entries.length}`); // 檢查每個 entry data.entries.forEach((entry, index) => { console.log(`\n 📝 條目 ${index + 1}:`); if (!entry.properties) { console.log(` ❌ 錯誤: 條目 ${index + 1} 缺少 properties`); hasErrors = true; return; } // 檢查必填欄位 const required = REQUIRED_FIELDS[data.database] || []; required.forEach(field => { if (!entry.properties[field]) { console.log(` ❌ 錯誤: 缺少必填欄位 "${field}"`); hasErrors = true; } else { // 取得標題內容 let content = ""; if (entry.properties[field].title) { content = entry.properties[field].title[0]?.text?.content || ""; } else if (entry.properties[field].rich_text) { content = entry.properties[field].rich_text[0]?.text?.content || ""; } console.log(` ✅ ${field}: "${content.substring(0, 40)}${content.length > 40 ? '...' : ''}"`); } }); // 檢查 select 欄位的值 if (VALID_SELECT_VALUES[data.database]) { Object.entries(entry.properties).forEach(([fieldName, fieldValue]) => { if (fieldValue.select && VALID_SELECT_VALUES[data.database][fieldName]) { const validValues = VALID_SELECT_VALUES[data.database][fieldName]; const actualValue = fieldValue.select.name; if (!validValues.includes(actualValue)) { console.log(` ⚠️ 警告: "${fieldName}" 的值 "${actualValue}" 可能無效`); console.log(` 有效值: ${validValues.join(", ")}`); warnings.push(`${fieldName} 值可能無效`); } } }); } }); } catch (error) { console.log(`❌ 錯誤: ${error.message}`); hasErrors = true; } // 總結 console.log("\n" + "=".repeat(60)); if (hasErrors) { console.log("❌ 檢查失敗 - 發現錯誤，請修正後再上傳"); return false; } else if (warnings.length > 0) { console.log(`⚠️ 檢查通過但有 ${warnings.length} 個警告`); warnings.forEach(w => console.log(` - ${w}`)); return true; } else { console.log("✅ 檢查通過 - 可以上傳！"); return true; } } // 主程式 function main() { const args = process.argv.slice(2); if (args.length === 0) { console.log("用法: node validate_json.js "); console.log("範例: node validate_json.js outputs/QuoteCards_batch1.json"); process.exit(1); } const filepath = args[0]; if (!fs.existsSync(filepath)) { console.log(`❌ 檔案不存在: ${filepath}`); process.exit(1); } const isValid = validateJSON(filepath); process.exit(isValid ? 0 : 1); } main();
